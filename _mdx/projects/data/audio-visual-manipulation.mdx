from pydub import AudioSegment

audio = AudioSegment.from_file("audio.mp3", format="mp3")

print(f"Audio length: {audio.duration_seconds} seconds")
print(f"Audio sample rate: {audio.frame_rate} Hz")
print(f"Audio channels: {audio.channels}")
print(f"Audio sample width: {audio.sample_width} bytes")

cropped_audio = audio[:30000]
cropped_audio.export("cropped_audio.mp3", format="mp3")
lowered_audio = audio.apply_gain(-6.0)
lowered_audio.export("lowered_audio.mp3", format="mp3")
fast_audio = audio.speedup(playback_speed=1.2)
fast_audio.export("fast_audio.mp3", format="mp3")

```
